return total; })(window); 然后保存，接着我们再建立一个dest文件夹最为输出的文件的目录，最后我们要让grunt来运行这个插件，所以需要在Gruntfile.js中配置： uglify:{ //uglify的更多配置查看官网 options:{ //这里的pkg 就是上面读取的package.json的内容 stripBanners:true, //banner，压缩后显示的描述 banner: '/*! - v- */', }, build:{ //这里可以使用src/*.js，表示src下的所有js都压缩混淆，还可以使用!*.min.js,表示排除 src:'src/test.js', dest:'dest/-.min.js', } } 上面这段配置加入到，grunt.initConfig({里面})，这个只是让grunt知道执行uglify的时候使用什么规则，具体的执行还需把这个插件加入到上面说的“任务”中代码如下： grunt.loadNpmTasks("grunt-contrib-uglify"); grunt.registerTask('default',['uglify']); 完整的配置如下： module.exports=function(grunt){ //grunt初始化配置信息 grunt.initConfig({ //获取 package.json的配置信息 pkg:grunt.file.readJSON('package.json'), //配置uglify的支持 uglify:{ //uglify的更多配置查看官网 options:{ //这里的pkg 就是上面读取的package.json的内容 stripBanners:true, //banner，压缩后显示的描述 banner: '/*! - v - */', }, dest:{ src:'src/test.js', dest:'dest/-.min.js', } } });
